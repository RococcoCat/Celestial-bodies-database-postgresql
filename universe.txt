CREATE DATABASE universe;
\c universe
CREATE TABLE galaxy(galaxy_id SERIAL UNIQUE NOT NULL, 
	name VARCHAR(40) UNIQUE NOT NULL,
	description TEXT,
	age_in_millions_of_years INT,
	galaxy_types VARCHAR(50),
	distance_from_earth NUMERIC,
	PRIMARY KEY(galaxy_id));
	
CREATE TABLE star(star_id SERIAL UNIQUE NOT NULL, 
	name VARCHAR(40) UNIQUE NOT NULL,
	description TEXT,
	age_in_millions_of_years INT,
	distance_from_earth NUMERIC,
	star_type VARCHAR(40),
	galaxy_id INT REFERENCES galaxy(galaxy_id),
	PRIMARY KEY(star_id));

CREATE TABLE planet(planet_id SERIAL UNIQUE NOT NULL, 
	name VARCHAR(40) UNIQUE NOT NULL,
	description TEXT,
	age_in_millions_of_years INT,
	distance_from_earth NUMERIC,
	planet_type VARCHAR(40),
	has_life BOOLEAN,
	has_water BOOLEAN,
	star_id INT REFERENCES star(star_id),
	PRIMARY KEY(planet_id));
	
CREATE TABLE moon(moon_id SERIAL UNIQUE NOT NULL, 
	name VARCHAR(40) UNIQUE NOT NULL,
	description TEXT,
	age_in_millions_of_years INT,
	distance_from_earth NUMERIC,
	planet_type VARCHAR(40),
	has_life BOOLEAN,
	has_water BOOLEAN,
	planet_id INT REFERENCES planet(planet_id),
	PRIMARY KEY(moon_id));
	
CREATE TABLE asteroid(asteroid_id SERIAL UNIQUE NOT NULL, 
	name VARCHAR(40) UNIQUE NOT NULL,
	description TEXT,
	age_in_millions_of_years INT,
	distance_from_earth NUMERIC,
	planet_type VARCHAR(40),
	has_life BOOLEAN,
	has_water BOOLEAN,
	PRIMARY KEY(asteroid_id));
